### 1. 声明
```js
// 与单引号效果一致
let str = `This is a string`
```
### 2. 内容可以出现换行符
```js
// 换行符会被保留
let str = `<ul>
             <li>Arthur</li>
             <li>Dutch</li>
          </ul>`
```
### 3. 变量拼接
```js
let name = 'Red Dead Redemption';
let series = `${name} is a good game!!`;
```
